555 inline void QXmlSimpleReader::next()
556 {
557  if ( !xmlRef.isEmpty() ) {
558  c = xmlRef[0];
559  xmlRef.remove( 0, 1 );
560  } else {
561  if ( c=='\n' || c=='\r' ) {
562  lineNr++;
563  columnNr = -1;
564  }
565  if ( pos >= xmlLength ) {
566  c = QEOF;
567  } else {
568  c = xml[pos];
569  columnNr++;
570  pos++;
571  }
572  }
573 }
